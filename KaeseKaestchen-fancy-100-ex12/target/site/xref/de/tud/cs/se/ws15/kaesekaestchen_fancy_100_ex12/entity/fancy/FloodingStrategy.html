8    * Implementation of Flooding. This event unmarks all edges in the row/column of
9    * the fancy edge.
10   */
11  public class FloodingStrategy implements FancyHandle {

13  	@Override
14  	public int action(Map gameMap, Player markingPlayer) {
15  		System.out.println("A devastating flood washes over the map");
16  		// id of the edge causing the action
17  		int fancyID = gameMap.getFancyId();
18  		// amount of columns
19  		int col = gameMap.getColumns();
20  		// amount of rows
21  		int row = gameMap.getRows();
22  		// all edges of the map
23  		Edge[] edges = gameMap.getEdges();
24  		// whether or not the edge causing the action is vertical
25  		boolean vertical = edges[fancyID].isVertical();
26  		// the total amount of edges
27  		int ec = gameMap.getEdgeCount();
28  		// the amount of fields the player closed with this action
29  		int closedFields = 0;
30  		// if some edge has id x, the next edge directly below has id x +
31  		// edgeIndexDiff
32  		int edgeIndexDiff = col + row + 1;
33  		// another index variable needed to count some rows
34  		int i = fancyID;
35  
36  		if (vertical) {
37  			// these two loops look unnecessary at first sight but it's a way of
38  			// avoiding to mark fancyID without a more complicated algorithm
39  			// mark every edge below
40  			i = fancyID + edgeIndexDiff;
41  			while (i < ec) {
42  				gameMap.undo(edges[i].getId());
43  				i += edgeIndexDiff;
44  			}
45  			// and every edge above
46  			i = fancyID - edgeIndexDiff;
47  			while (i >= 0) {
48  				gameMap.undo(edges[i].getId());
49  				i -= edgeIndexDiff;
50  			}
51  
52  		} else {
53  			// let j point to the last row but the same column
54  			int j = i;
55  			while (j + edgeIndexDiff < ec) {
56  				j += edgeIndexDiff;
57  			}
58  			// the columnIndex'th amount of columns is where fancyID is at
59  			// the leftmost columnIndex is 0, the rightmost col-1
60  			int columnIndex = col - (ec - j);
61  			// the index of the leftmost edge in this row
62  			int leftIndex = i - columnIndex;
63  			// iterate over every edge in this row except for the one causing
64  			// the fancy action
65  			for (int k = leftIndex; k < leftIndex + col; k++) {
66  				if (k != fancyID) {
67  					gameMap.undo(edges[k].getId());
68  				}
69  			}
70  		}
71  
72  		return closedFields;
73  	}
